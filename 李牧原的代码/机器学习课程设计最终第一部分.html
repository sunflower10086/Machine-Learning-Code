#利用库函数拟合模型
from sklearn.preprocessing import PolynomialFeatures
from sklearn.linear_model import LinearRegression
import numpy as np
import time
import matplotlib.pyplot as plt
# 构造数据
N=20
TN=50
x=np.linspace(-1,1,N).reshape((-1,1))
#分类的训练集 x1 y1
x1=np.linspace(-1,1,N-1).reshape((-1,1))
for i in range(len(x)-1):
    x1[i]=x[i]+1/N
y=np.sin(6*x).reshape((-1,1))
y1=np.sign(x1).reshape((-1,1))
#分类的测试集 tx ty1
tx=np.linspace(-1,1,TN).reshape((-1,1))
ty=np.sin(6*tx).reshape((-1,1))
ty1=np.sign(ty).reshape((-1,1))
start = time.time()
# 创建多项式特征回归
poly = PolynomialFeatures(degree=6)
x_poly = poly.fit_transform(x)
# 创建多项式特征分类
poly1 = PolynomialFeatures(degree=6)
x_poly1 = poly1.fit_transform(tx)
# 创建模型
mode = LinearRegression()
model = LinearRegression()
# 拟合模型回归
mode.fit(x_poly, y)
# 拟合模型分类
model.fit(x_poly1, ty)
# 预测
x_new =np.linspace(-1,1,TN).reshape((-1, 1))
x_new_poly = poly.fit_transform(x_new)
x_new_poly1 = poly.fit_transform(tx)

y_new = mode.predict(x_new_poly)
y_new1 = model.predict(x_new_poly1)
y_new2=np.sign(y_new1)
#回归的模型和实际比较
plt.plot(x,y,label='原函数图像')
plt.plot(x_new,y_new,label='训练模型的图像')
#分类的模型和实际比较
plt.scatter(tx,ty1,marker=".",label='正确的分类')
plt.scatter(tx,y_new2+0.015,marker=".",label='训练后分类')
end = time.time()
plt.text(-1,0,"用库函数实现")
plt.legend(loc="center right",prop= {'size': 7})

print("库函数运行时间:"+str(end-start))
